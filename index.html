Average treatment cost by hospital
Ranks hospitals from most affordable to most expensive
-- Average treatment cost per hospital SELECT h.hospital_name, h.city, h.state, ROUND(AVG(tc.cost), 0) AS avg_cost, COUNT(*) AS total_records FROM hospitals h JOIN treatment_costs tc ON h.hospital_id = tc.hospital_id GROUP BY h.hospital_id, h.hospital_name, h.city, h.state ORDER BY avg_cost ASC;
Cheapest hospital per treatment (CTE + RANK)
Uses RANK() window function to identify best-value provider
-- Cheapest hospital for each treatment WITH ranked_hospitals AS ( SELECT t.treatment_name, t.treatment_category, h.hospital_name, h.city, ROUND(AVG(tc.cost), 0) AS avg_cost, RANK() OVER ( PARTITION BY t.treatment_id ORDER BY AVG(tc.cost) ASC ) AS cost_rank FROM treatment_costs tc JOIN hospitals h ON tc.hospital_id = h.hospital_id JOIN treatments t ON tc.treatment_id = t.treatment_id GROUP BY t.treatment_id, h.hospital_id ) SELECT treatment_name, treatment_category, hospital_name, city, avg_cost FROM ranked_hospitals WHERE cost_rank = 1 ORDER BY avg_cost DESC;
Cost variance between hospitals
Identifies treatments with highest price spread
-- Cost variance per treatment SELECT t.treatment_name, t.treatment_category, MIN(tc.cost) AS min_cost, MAX(tc.cost) AS max_cost, (MAX(tc.cost) - MIN(tc.cost)) AS cost_variance, ROUND((MAX(tc.cost) - MIN(tc.cost)) / MIN(tc.cost) * 100, 1) AS variance_pct, COUNT(DISTINCT tc.hospital_id) AS num_hospitals FROM treatment_costs tc JOIN treatments t ON tc.treatment_id = t.treatment_id GROUP BY t.treatment_id, t.treatment_name, t.treatment_category ORDER BY cost_variance DESC LIMIT 10;
Treatment category analysis
Speciality-level cost metrics
-- Category-level cost analysis SELECT t.treatment_category, COUNT(DISTINCT t.treatment_id) AS treatment_count, COUNT(*) AS total_records, ROUND(AVG(tc.cost), 0) AS avg_cost, MIN(tc.cost) AS min_cost, MAX(tc.cost) AS max_cost, ROUND(STDDEV(tc.cost), 0) AS cost_stddev FROM treatment_costs tc JOIN treatments t ON tc.treatment_id = t.treatment_id GROUP BY t.treatment_category ORDER BY avg_cost DESC;
Duplicate removal
Using ROW_NUMBER() to detect and remove exact duplicates
-- Remove duplicates keeping the first occurrence DELETE FROM raw_treatment_costs WHERE record_id IN ( SELECT record_id FROM ( SELECT record_id, ROW_NUMBER() OVER ( PARTITION BY hospital_id, treatment_id, date_updated, cost ORDER BY record_id ) AS rn FROM raw_treatment_costs ) t WHERE rn > 1 );
Cost range validation
Flag and quarantine out-of-range cost records
-- Validate and flag each record SELECT record_id, hospital_name, treatment_name, cost, CASE WHEN cost IS NULL THEN 'MISSING_COST' WHEN cost <= 0 THEN 'INVALID_NEGATIVE' WHEN cost > 5000000 THEN 'EXCEEDS_THRESHOLD' ELSE 'VALID' END AS validation_flag FROM raw_treatment_costs WHERE cost IS NULL OR cost <= 0 OR cost > 5000000 ORDER BY validation_flag, hospital_name;

DDL — Create tables
CREATE TABLE hospitals ( hospital_id VARCHAR(10) PRIMARY KEY, hospital_name VARCHAR(100) NOT NULL UNIQUE, city VARCHAR(50) NOT NULL, state VARCHAR(50) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ) ENGINE=InnoDB; CREATE TABLE treatments ( treatment_id VARCHAR(10) PRIMARY KEY, treatment_name VARCHAR(150) NOT NULL UNIQUE, treatment_category VARCHAR(80) NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ) ENGINE=InnoDB; CREATE TABLE treatment_costs ( record_id VARCHAR(10) PRIMARY KEY, hospital_id VARCHAR(10) NOT NULL, treatment_id VARCHAR(10) NOT NULL, cost DECIMAL(12,2) NOT NULL, currency CHAR(3) DEFAULT 'INR', date_updated DATE NOT NULL, FOREIGN KEY (hospital_id) REFERENCES hospitals(hospital_id), FOREIGN KEY (treatment_id) REFERENCES treatments(treatment_id), INDEX idx_hospital (hospital_id), INDEX idx_treatment (treatment_id), INDEX idx_cost (cost) ) ENGINE=InnoDB;
